-- Arrastar botão de abrir (Mobile/PC)
do
    local dragging, dragStart, startPos, activeTouch
    OpenBtn.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            dragging = true; dragStart = input.Position; startPos = OpenBtn.Position
            if input.UserInputType == Enum.UserInputType.Touch then activeTouch = input.TouchId end
            input.Changed:Connect(function() if input.UserInputState == Enum.UserInputState.End then dragging = false; activeTouch = nil end end)
        end
    end)
    UIS.InputChanged:Connect(function(input)
        if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or (input.UserInputType == Enum.UserInputType.Touch and input.TouchId == activeTouch)) then
            local delta = input.Position - dragStart
            OpenBtn.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
        end
    end)
end

local function newSlider(parent, text, y, min, max, default, callback)
    local label = Instance.new("TextLabel", parent)
    label.Size = UDim2.new(0.9,0,0,20); label.Position = UDim2.new(0.05,0,0,y)
    label.Text = text .. ": " .. default; label.BackgroundTransparency = 1; label.TextColor3 = Color3.new(1,1,1)
    local sFrame = Instance.new("Frame", parent)
    sFrame.Size = UDim2.new(0.9,0,0,10); sFrame.Position = UDim2.new(0.05,0,0,y+25); sFrame.BackgroundColor3 = Color3.new(0.2,0.2,0.2)
    local knob = Instance.new("Frame", sFrame)
    knob.Size = UDim2.new(0, 15, 1, 4); knob.Position = UDim2.new((default-min)/(max-min), -7, 0, -2); knob.BackgroundColor3 = Color3.new(0,0.6,1)
    
    knob.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            local move; move = UIS.InputChanged:Connect(function(i)
                if i.UserInputType == Enum.UserInputType.MouseMovement or i.UserInputType == Enum.UserInputType.Touch then
                    local percent = math.clamp((i.Position.X - sFrame.AbsolutePosition.X) / sFrame.AbsoluteSize.X, 0, 1)
                    knob.Position = UDim2.new(percent, -7, 0, -2)
                    local val = math.floor(min + (max - min) * percent)
                    label.Text = text .. ": " .. val
                    callback(val)
                end
            end)
            input.Changed:Connect(function() if input.UserInputState == Enum.UserInputState.End then move:Disconnect() end end)
        end
    end)
    return y + 50
end

local function getRainbow() return Color3.fromHSV((tick() % 5) / 5, 1, 1) end

-- Sistema de Fly
RunService.RenderStepped:Connect(function()
    local char = LocalPlayer.Character
    if char and char:FindFirstChild("Humanoid") then
        char.Humanoid.WalkSpeed = Settings.WalkSpeed
        
        local root = char:FindFirstChild("HumanoidRootPart")
        if Settings.FlyEnabled and root then
            root.Velocity = Vector3.new(0,0.1,0)
            local moveDir = char.Humanoid.MoveDirection
            root.CFrame = root.CFrame + (moveDir * Settings.FlySpeed * 0.1)
        end
    end
end)

-- Aimbot e Hitbox Loop
RunService.RenderStepped:Connect(function()
    local rainbow = getRainbow()
    
    for _, plr in pairs(Players:GetPlayers()) do
        if plr ~= LocalPlayer and plr.Character and plr.Character:FindFirstChild("HumanoidRootPart") then
            local hrp = plr.Character.HumanoidRootPart
            
            -- Hitbox Visual RGB
            local visual = plr.Character:FindFirstChild("HitboxVisual")
            if Settings.HitboxEnabled then
                if not visual then
                    visual = Instance.new("Part", plr.Character)
                    visual.Name = "HitboxVisual"
                    visual.CastShadow = false
                    visual.CanCollide = false
                    visual.Anchored = true
                    visual.Transparency = 0.5
                    visual.Material = Enum.Material.Neon
                end
                visual.Color = rainbow
                visual.Size = Vector3.new(Settings.HitboxSize, Settings.HitboxSize, Settings.HitboxSize)
                visual.CFrame = hrp.CFrame
            elseif visual then 
                visual:Destroy() 
            end
        end
    end
end)

-- ESP Drawing Fix
local espTable = {}
local function makeESP(plr)
    local o = {line = Drawing.new("Line"), text = Drawing.new("Text")}
    o.line.Thickness = 1; o.text.Size = 14; o.text.Center = true; o.text.Outline = true
    espTable[plr] = o
end
Players.PlayerAdded:Connect(makeESP)
for _, p in pairs(Players:GetPlayers()) do if p ~= LocalPlayer then makeESP(p) end end

RunService.RenderStepped:Connect(function()
    local rainbow = getRainbow()
    for plr, o in pairs(espTable) do
        local char = plr.Character
        if Settings.ESPEnabled and char and char:FindFirstChild("HumanoidRootPart") and char:FindFirstChild("Humanoid") and char.Humanoid.Health > 0 then
            local hrp = char.HumanoidRootPart
            local pos, onScreen = Camera:WorldToViewportPoint(hrp.Position)
            
            o.line.Visible = Settings.ESPLines and onScreen
            o.text.Visible = Settings.ESPNames and onScreen
            
            if onScreen then
                local color = Settings.ESPRainbow and rainbow or Color3.new(1,1,1)
                o.line.From = Vector2.new(Camera.ViewportSize.X/2, Camera.ViewportSize.Y); o.line.To = Vector2.new(pos.X, pos.Y); o.line.Color = color
                o.text.Position = Vector2.new(pos.X, pos.Y - 20); o.text.Text = plr.Name; o.text.Color = color
            end
        else
            o.line.Visible = false; o.text.Visible = false
        end
    end
end)
