local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local UIS = game:GetService("UserInputService")
local LocalPlayer = Players.LocalPlayer
local Camera = workspace.CurrentCamera

local useDrawing = (type(Drawing) == "table")

-- Configurações (Mantendo sua base e adicionando novas)
local Settings = {
    HitboxEnabled = false, HitboxSize = 10,
    AimbotEnabled = false, AimbotSmoothing = 0.2, AimbotMaxDistance = 350, FOV = 120,
    ESPEnabled = false, ESPNames = true, ESPLines = true, ESPRainbow = true,
    Speed = 16, FlyEnabled = false, FlySpeed = 50
}

local ScreenGui = Instance.new("ScreenGui", LocalPlayer.PlayerGui)
ScreenGui.ResetOnSpawn = false

-- Botão de Abrir Móvel (Melhorado)
local OpenBtn = Instance.new("ImageButton", ScreenGui)
OpenBtn.Size = UDim2.fromOffset(55, 55)
OpenBtn.Position = UDim2.new(0.1, 0, 0.4, 0)
OpenBtn.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
OpenBtn.Image = "rbxassetid://6031068426"
local btnCorner = Instance.new("UICorner", OpenBtn); btnCorner.CornerRadius = UDim.new(1, 0)
local btnStroke = Instance.new("UIStroke", OpenBtn); btnStroke.Color = Color3.fromRGB(0, 150, 255); btnStroke.Thickness = 2

-- Arrastar Botão
do
    local dragging, dragStart, startPos
    OpenBtn.InputBegan:Connect(function(i)
        if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then
            dragging = true; dragStart = i.Position; startPos = OpenBtn.Position
            i.Changed:Connect(function() if i.UserInputState == Enum.UserInputState.End then dragging = false end end)
        end
    end)
    UIS.InputChanged:Connect(function(i)
        if dragging and (i.UserInputType == Enum.UserInputType.MouseMovement or i.UserInputType == Enum.UserInputType.Touch) then
            local delta = i.Position - dragStart
            OpenBtn.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
        end
    end)
end

local Tabs = {}
local function CreateTab(name, order)
    local frame = Instance.new("ScrollingFrame", Container)
    frame.Size = UDim2.new(1, 0, 1, 0); frame.BackgroundTransparency = 1; frame.Visible = (order == 1)
    frame.CanvasSize = UDim2.new(0, 0, 1.5, 0); frame.ScrollBarThickness = 2
    Tabs[name] = frame
    
    local btn = Instance.new("TextButton", TabHolder)
    btn.Size = UDim2.new(1, -10, 0, 35); btn.Position = UDim2.new(0, 5, 0, (order-1)*40 + 10)
    btn.Text = name; btn.BackgroundColor3 = Color3.fromRGB(40, 40, 40); btn.TextColor3 = Color3.new(1, 1, 1)
    btn.Font = "GothamBold"; Instance.new("UICorner", btn)
    
    btn.MouseButton1Click:Connect(function()
        for _, t in pairs(Tabs) do t.Visible = false end
        frame.Visible = true
    end)
end

local function NewToggle(parent, text, settingKey)
    local btn = Instance.new("TextButton", parent)
    btn.Size = UDim2.new(1, -10, 0, 38)
    btn.BackgroundColor3 = Color3.fromRGB(150, 50, 50)
    btn.Text = text .. ": OFF"; btn.TextColor3 = Color3.new(1, 1, 1); btn.Font = "GothamBold"
    Instance.new("UICorner", btn)
    
    local function update()
        local state = Settings[settingKey]
        btn.Text = text .. (state and ": ON" or ": OFF")
        btn.BackgroundColor3 = state and Color3.fromRGB(50, 150, 50) or Color3.fromRGB(150, 50, 50)
    end
    
    btn.MouseButton1Click:Connect(function()
        Settings[settingKey] = not Settings[settingKey]
        update()
    end)
    Instance.new("UIListLayout", parent).Padding = UDim.new(0, 5)
    return btn
end

CreateTab("Main", 1); CreateTab("ESPs", 2); CreateTab("Troll", 3)

NewToggle(Tabs.Troll, "Fly Mode", "FlyEnabled")
-- Função de auxílio: RGB
local function getRGB() return Color3.fromHSV((tick() % 5) / 5, 1, 1) end

-- Loop Principal (Hitbox e Fly)
RunService.RenderStepped:Connect(function()
    local char = LocalPlayer.Character
    local rainbow = getRGB()

    -- HITBOX VISUAL (Apenas para você)
    if Settings.HitboxEnabled then
        for _, plr in pairs(Players:GetPlayers()) do
            if plr ~= LocalPlayer and plr.Character and plr.Character:FindFirstChild("HumanoidRootPart") then
                local hrp = plr.Character.HumanoidRootPart
                local visual = plr.Character:FindFirstChild("HitboxVisual")
                if not visual then
                    visual = Instance.new("BoxHandleAdornment", plr.Character)
                    visual.Name = "HitboxVisual"; visual.AlwaysOnTop = true; visual.ZIndex = 10
                    visual.Adornee = hrp; visual.Transparency = 0.5
                end
                visual.Color3 = rainbow
                visual.Size = Vector3.new(Settings.HitboxSize, Settings.HitboxSize, Settings.HitboxSize)
                hrp.Size = Vector3.new(Settings.HitboxSize, Settings.HitboxSize, Settings.HitboxSize)
                hrp.CanCollide = false
            end
        end
    end

    -- FLY ANALÓGICO
    if char and char:FindFirstChild("HumanoidRootPart") then
        local hrp = char.HumanoidRootPart
        if Settings.FlyEnabled then
            local moveDir = char.Humanoid.MoveDirection
            local look = Camera.CFrame.LookVector
            local right = Camera.CFrame.RightVector
            hrp.Velocity = (look * moveDir.Z * Settings.FlySpeed) + (right * moveDir.X * Settings.FlySpeed) + Vector3.new(0, 0.5, 0)
        end
    end
end)

-- ESP CORRIGIDO (Agora realmente desliga se o botão for OFF)
local espTable = {}
local function createESP(plr)
    if not useDrawing then return end
    local o = {line = Drawing.new("Line"), text = Drawing.new("Text")}
    o.line.Thickness = 1; o.text.Size = 14; o.text.Center = true; o.text.Outline = true
    espTable[plr] = o
end

Players.PlayerAdded:Connect(createESP)
for _, p in pairs(Players:GetPlayers()) do if p ~= LocalPlayer then createESP(p) end end

RunService.RenderStepped:Connect(function()
    local color = Settings.ESPRainbow and getRGB() or Color3.new(1,1,1)
    for plr, o in pairs(espTable) do
        local char = plr.Character
        -- CONDIÇÃO DE ATIVAÇÃO: Somente se ESPEnabled for TRUE
        if Settings.ESPEnabled and char and char:FindFirstChild("HumanoidRootPart") and char.Humanoid.Health > 0 then
            local hrp = char.HumanoidRootPart
            local pos, onScreen = Camera:WorldToViewportPoint(hrp.Position)
            
            o.line.Visible = Settings.ESPLines and onScreen
            o.text.Visible = Settings.ESPNames and onScreen
            
            if onScreen then
                o.line.From = Vector2.new(Camera.ViewportSize.X/2, Camera.ViewportSize.Y)
                o.line.To = Vector2.new(pos.X, pos.Y); o.line.Color = color
                o.text.Position = Vector2.new(pos.X, pos.Y - 20)
                o.text.Text = plr.Name .. " [" .. math.floor(char.Humanoid.Health) .. "HP]"
                o.text.Color = color
            end
        else
            o.line.Visible = false
            o.text.Visible = false
        end
    end
end)
